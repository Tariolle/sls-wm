\section{Motivation}

\begin{frame}{Why World Models?}
  \vfill
  \begin{itemize}
    \item Model-free RL $\to$ \textbf{millions of interactions}
    \item World Models: learned environment, training \textbf{in imagination}
    \item \textbf{Geometry Dash}: deterministic, binary action, precision timing
    \item \textbf{Real-time}: 30 FPS $\to$ reaction in $<$33ms
    \item \textbf{Goal}: master unseen levels, trained \emph{entirely} in imagination
  \end{itemize}
  \vfill
\end{frame}

\section{Phase 1: Vision}

\begin{frame}{Vision Model: $\beta$-VAE $\to$ VQ-VAE $\to$ FSQ-VAE}
  \begin{itemize}
    \item \textbf{$\beta$-VAE}: Gaussian posterior $\Rightarrow$ always blurry. No spatial structure.
    \item \textbf{VQ-VAE}: discrete, spatially localized tokens $\Rightarrow$ sharp. But codebook collapse, commitment cost tuning.
    \item \textbf{FSQ-VAE} (Mentzer et al., 2023): round to $L$ fixed levels. No codebook collapse, no auxiliary losses, 100\% utilization by construction.
  \end{itemize}

  \vspace{0.3em}
  \begin{center}
    Levels $[8,5,5,5]$ $\Rightarrow$ \fsqcodes{} codes\\
    $\spatialres\times\spatialres$ grid $=$ \textbf{\ntokens{} tokens/frame}
  \end{center}
  % TODO: add before/after reconstruction comparison
\end{frame}

  \vspace{0.3em}
  \begin{itemize}
    \item \textbf{MSE} reconstruction on $(\mathbf{x}_t, \mathbf{x}_{t+1})$ pairs (STE through quantizer)
    \item \textbf{GRWM slowness}: penalizes large $\|\mathbf{z}_e^{t+1}-\mathbf{z}_e^t\|^2$ -- temporally smooth latents
    \item \textbf{GRWM uniformity}: spreads $\mathbf{z}_e$ across the FSQ hypercube -- avoids dead codes
    \item Trained \emph{independently} of M -- frozen tokens fed to the transformer afterwards
  \end{itemize}
\end{frame}

\section{Phase 2: Memory}
% ==========================================================

\begin{frame}{FSQ-Structured Label Smoothing}
  \vspace{0.5em}
  \textbf{Problem}: standard CE assigns zero mass to every non-target token: a near-miss neighbour is penalised as harshly as a random code.

  \vspace{0.5em}
  \textbf{SLS}: replace the one-hot target with a narrow Gaussian centered on the correct code in FSQ coordinate space ($\sigma = 1.0$). Near neighbours get small positive mass; distant codes get none.

  \vspace{0.5em}
  Rewards near-misses without diluting confidence in the correct code.

\end{frame}

\begin{frame}{Training: FSQ-Neighbor Token Noise}
  \vspace{1.2em}
  \begin{columns}[T]
    \column{0.55\textwidth}
    Noise on context tokens during training: with probability 5\% per token, replace it by a $\pm 1$ neighbour in one FSQ dimension.

    \vspace{0.5em}
    \begin{itemize}
      \item Visually near-identical patches (one quantization step)
      \item Forces the transformer to treat near-miss codes as equivalent context
      \item Matches the SLS assumption that near-miss is almost-right: consistent signal on both sides of the loss
    \end{itemize}

    \column{0.45\textwidth}
    \vspace{-1.7em}
    \begin{center}
      \includegraphics[width=0.80\textwidth]{fsq_neighbors.png}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Transformer: Training}
  \begin{itemize}
    \item AdamW, LR $10^{-3}$, weight decay $0.05$, dropout $0.1$, batch 256, 200 epochs
    \item Cosine LR + linear warmup, early stopping on val loss
    \item Death oversampling $15\times$
    \item Token noise: 5\% random + 5\% FSQ-neighbour substitution
    \item Weight tying between input embedding and output head
    \item Best epoch 146, val acc 35.6\%, death F1 0.78
  \end{itemize}
\end{frame}

\section{Analysis}
% ==========================================================

\begin{frame}{What Worked}
  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{FSQ-VAE}: sharp, 100\% utilization, no tuning
    \item \textbf{Sobel}: filters visual noise $\to$ structural perception
    \item \textbf{Vertical shifts}: massively reduced overfitting, physically valid
    \item \textbf{Structured label smoothing}: narrow Gaussian on FSQ lattice distance
    \item \textbf{Action tokens in the sequence}: simple, stable, no special init tricks
    \item \textbf{Scaling to 384d}: visible jump on every metric (acc, death F1, BC, real-game)
    \item \textbf{AC-CPC} ($w=0.1$): better dream quality at no cost
  \end{itemize}
\end{frame}

\begin{frame}{What Didn't Work (or Didn't Help)}
  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{$\beta$-VAE}: blurry, large grid search $\to$ abandoned
    \item \textbf{Pure RL (CMA-ES)}: $\sim$12 steps (local optimum) $\to$ BC required
    \item \textbf{MaskGIT}: degraded on 64 tokens, parallel single-step better
    \item \textbf{Mask scheduling during training}: train/inference mismatch, fixed in V2
    \item \textbf{Soft continuation gating}: agent gamed airtime
    \item \textbf{PMPO}: oscillated with binary action space
  \end{itemize}
\end{frame}

\begin{frame}{Tried After V3 -- Did Not Outperform Yet}
  \vspace{0.3em}
  More elegant, faster to train, but less stable / less precise on the real game:

  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{End-to-end V+M training} (JEPA way) -- V and M trained jointly with only \emph{two} losses (MSE on pixels + CE on tokens). Stable, but the resulting FSQ is harder to act on than V3's separately-trained FSQ.
    \item \textbf{AdaLN-Zero action conditioning}: actions removed from the sequence, injected via AdaLN. Best CPC ever, but 5 runs diverged before QK-norm and gate-bias-1 init unblocked it.
    \item \textbf{Scaling to 512d}: bigger, slower, no real-game gain on the new latents.
    \item \textbf{MLPPolicy on $\mathbf{h}_t$ only}: 2$\times$ faster convergence, smallest train/eval gap, but the V3 CNN still wins on real-game with the V3 FSQ.
    \item \textbf{Multi-token prediction}: had a silent off-by-one bug, dropped in V4.
  \end{itemize}

  \vspace{0.3em}
  \begin{center}
    \small
    Net: V3 stays the best deployment despite V4--V6 being more elegant on paper.
  \end{center}
\end{frame}

\section{Limitations}
% ==========================================================

\begin{frame}{Limitations \& Future Work}
  \begin{itemize}
    \item \textbf{Sample efficiency}: trained on only 2h20 of gameplay -- world models make this tractable
    \item \textbf{Dream horizon}: dreams stay coherent for tens to hundreds of frames, then hallucinate -- repeated obstacles, impassable sections, or empty levels
    \item \textbf{Scope}: trained on simple mechanics (portals, jump pads, orbs, standard gravity). Complex levels with noisy backgrounds and harder dynamics are the natural next step.
  \end{itemize}
\end{frame}

\begin{frame}{Transformer: Sequence Layout}
  \vspace{0.5em}
  Each frame block has \textbf{65 positions}: 64 visual tokens + 1 status token (ALIVE / DEATH).
  Action tokens are interleaved between frame blocks. The target frame is replaced by a learnable mask embedding -- no ground-truth leakage.

    \node[ablk, right=0.1cm of f1] (a1) {$a_1$};
    \node[fblk, right=0.1cm of a1] (f2) {$f_2$ (65)};
    \node[ablk, right=0.1cm of f2] (a2) {$a_2$};

  \vspace{0.3em}
  \begin{itemize}
    \item \textbf{Block-causal} attention: bidirectional within each block, causal across.
    \item \textbf{3D-RoPE}: head dim split into row / col / frame bands, lower $\theta$ for spatial axes.
    \item \textbf{Parallel decode}: all 65 target positions predicted in one forward pass.
  \end{itemize}

\begin{frame}{Transformer: Auxiliary Contrastive Loss (AC-CPC)}
  \vspace{0.3em}
  Beyond next-frame CE, the transformer is trained with an \textbf{action-conditional contrastive predictive coding} loss (TWISTER, Burchi \& Timofte 2025):

  \vspace{0.5em}
  \begin{itemize}
    \item For each $k\in\{1,\ldots,K\}$, predict $\mathbf{h}_{t+k}$ from $\mathbf{h}_t$ + $a_{t:t+k-1}$ via an MLP head.
    \item InfoNCE loss with in-batch negatives (temperature 0.1).
    \item Forces $\mathbf{h}_t$ to encode action-conditional dynamics, not just static appearance.
  \end{itemize}

  \vspace{0.5em}
  Combined loss: $\mathcal{L} = \mathcal{L}_{\text{CE+SLS}} + 0.1 \cdot \mathcal{L}_{\text{AC-CPC}}$.
\end{frame}
