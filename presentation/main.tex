\section{Motivation}
% ==========================================================

\begin{frame}{Why World Models?}
  \vfill
  \begin{itemize}
    \item Model-free RL $\to$ \textbf{millions of interactions}
    \item World Models: learned environment, training \textbf{in imagination}
    \item \textbf{Geometry Dash}: deterministic, binary action, precision timing
    \item \textbf{Real-time}: 30 FPS $\to$ no MCTS, reaction $<$33ms
    \item \textbf{Goal}: master unseen levels, trained entirely \emph{in imagination}
  \end{itemize}
  \vfill
\end{frame}

  \textbf{Problem}: $\sim$252K frames is limited data $\Rightarrow$ overfitting risk

  \vspace{0.3em}
  \textbf{Solution}: apply vertical pixel shifts at the dataset level, used across all training phases

  \begin{center}
  \resizebox{0.85\textwidth}{!}{%

  \begin{itemize}
    \item \nshifts{} vertical shifts $\times$ \nbaseepisodes{} episodes $=$ \textbf{\naugeps{} augmented episodes}
    \item Vertical shifts simulate different stage elevations (physically valid)
    \item Horizontal shifts removed: player X position is fixed in Geometry Dash
    \item Edge padding to fill shifted borders
  \end{itemize}
\end{frame}

\section{Phase 1: V (FSQ-VAE)}
% ==========================================================

\begin{frame}{Vision Model: $\beta$-VAE $\to$ VQ-VAE $\to$ FSQ-VAE}
  \begin{itemize}
    \item \textbf{$\beta$-VAE}: Gaussian posterior $\Rightarrow$ always blurry. No spatial structure.
    \item \textbf{VQ-VAE}: discrete, spatially localized tokens $\Rightarrow$ sharp. But codebook collapse, commitment cost tuning.
    \item \textbf{FSQ-VAE} (Mentzer et al., 2023): round to $L$ fixed levels. No codebook, no auxiliary losses, 100\% utilization by construction.
  \end{itemize}

  \vspace{0.3em}
  \begin{center}
    Levels $[8,5,5,5]$ $\Rightarrow$ \fsqcodes{} codes\\
    $\spatialres\times\spatialres$ grid $=$ \textbf{\ntokens{} tokens/frame}
  \end{center}
  % TODO: add before/after reconstruction comparison
\end{frame}

\begin{frame}{FSQ-VAE: Training}
  \begin{itemize}
    \item $\sim$252K frames ($\sim$\naugeps{} episodes), batch 2048
    \item GRWM regularization (temporal slowness + uniformity)
    \item Cosine LR schedule, 200 epochs
    \item GPU: A100, total training time: $\sim$48 min
  \end{itemize}
\end{frame}

\begin{frame}{Training: Dual Token Noise}
  \vspace{0.5em}
  \begin{columns}[T]
    \column{0.55\textwidth}
    Two complementary noise types on context tokens:

    \vspace{0.5em}
    \textbf{1. Random replacement (5\%)}
    \begin{itemize}
      \item Any of 1000 tokens uniformly
      \item Forces global robustness
    \end{itemize}

    \vspace{0.5em}
    \textbf{2. FSQ neighbor substitution (5\%)} {\color{insared}\textit{(novel)}}
    \begin{itemize}
      \item $\pm1$ in one FSQ dimension
      \item Visually near-identical patches
      \item Forces embedding to learn codebook topology
    \end{itemize}

    \column{0.45\textwidth}
    \vspace{-1.2em}
    \begin{center}
      \includegraphics[width=0.80\textwidth]{fsq_neighbors.png}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}[t]{Novel: FSQ-Structured Label Smoothing}
  \textbf{Problem}: uniform smoothing $\to$ FSQ neighbor = random token

  \vspace{0.5em}
  \textbf{Solution} {\color{insared}\textit{(novel)}}: Gaussian kernel over FSQ coordinate distance:
  \[
    w(a,b) = \exp\!\left(-\frac{d^2(a,b)}{2\sigma^2}\right), \quad d^2 = \sum_i (\Delta_i)^2, \quad \sigma = 0.9
  \]

  $\sigma$ calibrated from empirical MSE analysis (800 samples per perturbation):

  \vspace{0.2em}
  \begin{center}
    \small
    \begin{tabular}{lcc}
      \toprule
      Perturbation & Patch MSE & Ratio \\
      \midrule
      $\pm1$ single dim & 0.000078 & 1.0$\times$ \\
      $+1$ two dims & 0.000200 & 2.55$\times$ \\
      $+2$ single dim & 0.000493 & \textbf{6.30$\times$} \\
      \bottomrule
    \end{tabular}
  \end{center}

  \vspace{0.2em}\end{frame}

\begin{frame}{Controller: Training Protocol}
  \vspace{0.5em}
  \textbf{Stage 1: Behavioral Cloning} (initialization)
  \begin{itemize}
    \item Death + expert episodes, class-weighted BCE (1.5$\times$ jump weight)
    \item Feed context through world model $\to$ get $\mathbf{h}_t$, predict expert action
  \end{itemize}

  \vspace{0.5em}
  \textbf{Stage 2: PPO in Dream} (generalization)
  \begin{itemize}
    \item Uniform context sampling, excluding last $2K$ frames per episode
    \item Reward: $+1$ per step survived $-0.2$ per jump, rollout stops at death prediction
    \item 512 episodes/iteration, 45-step rollouts, 4 PPO epochs, GAE ($\gamma$=0.995, $\lambda$=0.95)
    \item Clipped surrogate + percentile advantage norm + EMA target critic
  \end{itemize}

  \vspace{0.5em}
  \textbf{Test:} zero-shot deployment on \emph{unseen} levels
\end{frame}

\section{Analysis}
% ==========================================================

\begin{frame}{What Worked}
  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{FSQ-VAE}: sharp, 100\% utilization, no tuning
    \item \textbf{Sobel}: filters visual noise $\to$ structural perception
    \item \textbf{Vertical shifts}: reduced overfitting, physically valid
    \item \textbf{Parallel decoding}: 1 pass vs 65 $\to$ real-time
    \item \textbf{Structured label smoothing}: grounded in FSQ distance analysis
    \item \textbf{AdaLN-Zero}: best CPC (0.133)
    \item \textbf{QK-norm}: stabilized AdaLN (4 runs diverged without)
  \end{itemize}
\end{frame}

\begin{frame}{What Didn't Work (or Didn't Help)}
  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{$\beta$-VAE}: blurry, large grid search $\to$ abandoned
    \item \textbf{LR $10^{-3}$ w/o augmentation}: 13.5pp train/val gap $\to$ memorization
    \item \textbf{FSQ neighbor substitution}: no measurable impact, kept (free)
    \item \textbf{Pure RL (CMA-ES)}: $\sim$12 steps (local optimum) $\to$ BC required
    \item \textbf{Death oversample 15$\times$}: memorization (44pp precision gap). Fix: 5$\times$ + expert
    \item \textbf{MaskGIT}: degraded on 64 tokens, parallel single-step better
    \item \textbf{CNN controller on $\mathbf{z}_t$}: duplicated work the transformer already did. Pure MLP on $\mathbf{h}_t$ trains 2-5$\times$ faster and generalizes better.
  \end{itemize}
\end{frame}

% ==========================================================
\section{Limitations}
% ==========================================================

\begin{frame}{Limitations \& Future Work}
  \begin{itemize}
    \item \textbf{Controller precision}: jumps too often/early
    \item \textbf{Dream/reality gap}: $\mathbf{h}_t$ distribution differs in real game
    \item \textbf{Sobel bottleneck}: solved with GPU (7ms vs 10ms CPU)
    \item \textbf{Token-level augmentation}: impossible post-tokenization
  \end{itemize}
\end{frame}
