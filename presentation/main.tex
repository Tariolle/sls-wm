\section{Motivation}
% ==========================================================

\begin{frame}{Why World Models?}
  \vfill
  \begin{itemize}
    \item Model-free RL $\to$ \textbf{millions of interactions}
    \item World Models: learned environment, training \textbf{in imagination}
    \item \textbf{Geometry Dash}: deterministic, binary action, precision timing
    \item \textbf{Real-time}: 30 FPS $\to$ reaction in $<$33ms
    \item \textbf{Goal}: master unseen levels, trained \emph{entirely} in imagination
  \end{itemize}
  \vfill
\end{frame}

\section{Phase 1: World Model}

\begin{frame}{Vision Model: $\beta$-VAE $\to$ VQ-VAE $\to$ FSQ-VAE}
  \begin{itemize}
    \item \textbf{$\beta$-VAE}: Gaussian posterior $\Rightarrow$ always blurry. No spatial structure.
    \item \textbf{VQ-VAE}: discrete, spatially localized tokens $\Rightarrow$ sharp. But codebook collapse, commitment cost tuning.
    \item \textbf{FSQ-VAE} (Mentzer et al., 2023): round to $L$ fixed levels. No codebook collapse, no auxiliary losses, 100\% utilization by construction.
  \end{itemize}

  \vspace{0.3em}
  \begin{center}
    Levels $[5,5,5,5]$ $\Rightarrow$ \fsqcodes{} codes\\
    $\spatialres\times\spatialres$ grid $=$ \textbf{\ntokens{} tokens/frame}
  \end{center}
  % TODO: add before/after reconstruction comparison
\end{frame}

\begin{frame}{FSQ-VAE: Training}
  \begin{itemize}
    \item $\sim$252K frames ($\sim$\naugeps{} episodes), batch 1024
    \item \textbf{Jointly} optimized with the transformer: no frozen pre-stage
    \item Pixel MSE kept as an \textbf{anchor} on codebook semantics
    \item Prediction gradients route back into the encoder via a straight-through estimator (STE) through a learnable linear projection \texttt{fsq\_grad\_proj}
    \item \textbf{Single-group optimizer} (encoder + transformer share one AdamW group, LR $10^{-3}$, weight decay $10^{-3}$) + \textbf{single global grad clip at 10.0} (sized to the recon-dominated encoder gradient scale)
    \item Cosine LR schedule + linear warmup, 200 epochs
  \end{itemize}
\end{frame}

\begin{frame}{Training: FSQ-Neighbor Token Noise}
  \vspace{0.5em}
  \begin{columns}[T]
    \column{0.55\textwidth}
    Noise on context tokens during training: with probability 5\% per token, replace it by a $\pm 1$ neighbour in one FSQ dimension.

    \vspace{0.5em}
    \begin{itemize}
      \item Visually near-identical patches (one quantization step)
      \item Forces the transformer to treat near-miss codes as equivalent context
      \item Matches the SLS assumption that near-miss is almost-right: consistent signal on both sides of the loss
    \end{itemize}
    
    \column{0.45\textwidth}
    \vspace{-1.2em}
    \begin{center}
      \includegraphics[width=0.80\textwidth]{fsq_neighbors.png}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}[t]{FSQ-Structured Label Smoothing}
  \vspace{0.4em}
  \textbf{Problem}: uniform smoothing treats an FSQ neighbour like a random token.

  \vspace{0.3em}
  \textbf{Kernel study} over FSQ coordinate distance $d = \lVert \mathbf{c}_i - \mathbf{c}_j \rVert$:
  \vspace{-0.2em}
  \[
    k_{\text{Gauss}}(d) = e^{-d^2/2\sigma^2}, \quad
    k_{\text{Laplace}}(d) = e^{-d/\sigma}, \quad
    k_{\text{Cauchy}}(d) = \frac{1}{1 + d^2/\sigma^2}
  \]

  \vspace{0.2em}
  \textbf{Design choice}: narrow Gaussian ($\sigma = 0.7$). Gaussian's squared-exponential tails decay faster than Laplace or Cauchy, so distant neighbours receive negligible mass. The goal is to reward near-misses without tempting the predictor to commit less strongly to the correct code.

\end{frame}

\begin{frame}{Controller: Training Protocol}
  \vspace{0.5em}
  \textbf{Stage 1: Behavioral Cloning} (initialization)
  \begin{itemize}
    \item Death + expert episodes, class-weighted BCE (1.5$\times$ jump weight)
    \item Feed context through world model $\to$ get $\mathbf{h}_t$, predict expert action
  \end{itemize}

  \vspace{0.5em}
  \textbf{Stage 2: PPO in Dream} (generalization)
  \begin{itemize}
    \item Reward: $+1$ per step survived $-0.5$ per jump, rollout stops at death prediction
    \item 2048 episodes/iteration, 25-step rollouts, 4 PPO epochs, GAE ($\gamma$=0.995, $\lambda$=0.95)
    \item Clipped surrogate + percentile advantage norm + EMA target critic
  \end{itemize}

  \vspace{0.5em}
  \textbf{Test:} zero-shot deployment on the real game (no fine-tuning)

\section{Analysis}
% ==========================================================

\begin{frame}{What Worked}
  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{FSQ-VAE}: sharp, 100\% utilization, no tuning
    \item \textbf{Sobel}: filters visual noise $\to$ structural perception
    \item \textbf{Vertical shifts}: massively reduced overfitting, physically valid
    \item \textbf{Parallel decoding}: 1 pass vs 65 $\to$ real-time
    \item \textbf{Structured label smoothing}: narrow Gaussian on FSQ lattice distance
    \item \textbf{AdaLN-Zero}: action conditioning from DiT
    \item \textbf{QK-norm}: stabilized AdaLN (4 runs diverged without)
  \end{itemize}
\end{frame}

\begin{frame}{What Didn't Work (or Didn't Help)}
  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{$\beta$-VAE}: blurry, large grid search $\to$ abandoned
    \item \textbf{Pure RL (CMA-ES)}: $\sim$12 steps (local optimum) $\to$ BC required
    \item \textbf{MaskGIT}: degraded on 64 tokens, parallel single-step better
    \item \textbf{CNN controller on $\mathbf{z}_t$}: duplicated work the transformer already did
    \item \textbf{Pure-JEPA} (no pixel anchor): codebook collapse; our marginal anti-collapse regularizer (CvM adapted from LeWorldModel's SIGReg) was insufficient for joint-distribution degeneracy
  \end{itemize}
\end{frame}

% ==========================================================
\section{Limitations}
% ==========================================================

\begin{frame}{Limitations \& Future Work}
  \begin{itemize}
    \item \textbf{Scope}: only one game (and standard levels). Generalization to other games/levels unknown
    \item \textbf{Sample efficiency}: $\sim$250K frames (only 2h20 of gameplay) is extremely small for deep learning
  \end{itemize}
\end{frame}
